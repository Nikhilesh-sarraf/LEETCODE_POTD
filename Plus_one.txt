class Solution {
public:
    vector<int> plusOne(vector<int>& digits) {

        int n = digits.size();
        int carry = 1;

        // Start from the last digit
        for(int i = n - 1; i >= 0; i--) {
            int sum = digits[i] + carry;
            digits[i] = sum % 10;
            carry = sum / 10;

            if(carry == 0)
                break;
        }

    
        if(carry) {
            digits.insert(digits.begin(), 1);
        }

        return digits;
    }
};
